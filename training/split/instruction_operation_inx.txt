# 6502 Microprocessor - INX pseudocode: XR = (XR + 1) & 0xff; update Sign and Zero.

/* INX */
    unsigned src = XR;
    src = (src + 1) & 0xff;
    SET_SIGN(src);
    SET_ZERO(src); 
    XR = (src);
