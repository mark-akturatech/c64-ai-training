# 6502 Microprocessor - AND pseudocode: src &= AC; set sign and zero; AC = result. Bitwise AND operation affecting N and Z flags.

/* AND */
    src &= AC;
    SET_SIGN(src);
    SET_ZERO(src);
    AC = src;
