# 6502 Microprocessor - BPL pseudocode: branch if Sign flag clear; adjust cycles for page crossing then PC = relative address.

/* BPL */
    if (!IF_SIGN()) {
	clk += ((PC & 0xFF00) != (REL_ADDR(PC, src) & 0xFF00) ? 2 : 1);
	PC = REL_ADDR(PC, src);
    }
