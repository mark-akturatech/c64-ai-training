# 6502 Microprocessor - DEX pseudocode: XR = (XR - 1) & 0xff; update Sign and Zero; store back to X register.

/* DEX */
    unsigned src = XR;
    src = (src - 1) & 0xff;
    SET_SIGN(src);
    SET_ZERO(src);
    XR = (src);
