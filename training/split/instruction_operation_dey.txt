# 6502 Microprocessor - DEY pseudocode: decrement Y register, update Sign and Zero.

/* DEY */
    unsigned src = YR;
    src = (src - 1) & 0xff;
    SET_SIGN(src);
    SET_ZERO(src);
    YR = (src);
