# 6502 Microprocessor - BNE pseudocode: branch if Zero flag clear; adjust cycles for page crossing then PC = relative address.

/* BNE */
    if (!IF_ZERO()) {
	clk += ((PC & 0xFF00) != (REL_ADDR(PC, src) & 0xFF00) ? 2 : 1);
	PC = REL_ADDR(PC, src);
    }
