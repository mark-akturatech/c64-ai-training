.,FD02 A2 05    LDX #$05        A0INT  LDX #TBLA0E-TBLA0R ;CHECK FOR $8000
.,FD04 BD 0F FD LDA $FD0F,X     A0IN1  LDA TBLA0R-1,X

.,FD0A D0 03    BNE $FD0F              BNE A0IN2
.,FD0C CA       DEX                    DEX
.,FD0D D0 F5    BNE $FD04              BNE A0IN1
.,FD0F 60       RTS             A0IN2  RTS
