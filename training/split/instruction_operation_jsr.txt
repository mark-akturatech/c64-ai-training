# 6502 Microprocessor - JSR pseudocode: decrement PC, push return address (high/low) to stack, set PC to target.

/* JSR */
    PC--;
    PUSH((PC >> 8) & 0xff);	/* Push return address onto the stack. */
    PUSH(PC & 0xff);
    PC = (src);
