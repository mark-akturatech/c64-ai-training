# 6502 Microprocessor - BMI pseudocode: branch if Sign flag set; account for page crossing cycles and set PC accordingly.

/* BMI */
    if (IF_SIGN()) {
	clk += ((PC & 0xFF00) != (REL_ADDR(PC, src) & 0xFF00) ? 2 : 1);
	PC = REL_ADDR(PC, src);
    }
