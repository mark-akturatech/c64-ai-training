# 6502 Microprocessor - RTS pseudocode: pull return address low/high from stack, add 1, set PC to that value.

/* RTS */
    src = PULL();
    src += ((PULL()) << 8) + 1;	/* Load return address from stack and add 1. */
    PC = (src);
