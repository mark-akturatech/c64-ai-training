# 6502 Microprocessor - BCS pseudocode: if Carry flag set then handle page crossing cycles and PC = relative address.

/* BCS */
    if (IF_CARRY()) {
	clk += ((PC & 0xFF00) != (REL_ADDR(PC, src) & 0xFF00) ? 2 : 1);
	PC = REL_ADDR(PC, src);
    }
