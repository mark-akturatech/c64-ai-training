# 6502 Microprocessor - BCC pseudocode: if Carry flag clear then add cycles for page crossing and set PC to relative address.

/* BCC */
    if (!IF_CARRY()) {
	clk += ((PC & 0xFF00) != (REL_ADDR(PC, src) & 0xFF00) ? 2 : 1);
	PC = REL_ADDR(PC, src);
    }
