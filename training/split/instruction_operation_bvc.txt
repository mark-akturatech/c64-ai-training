# 6502 Microprocessor - BVC pseudocode: branch if Overflow flag clear; adjust cycles for page crossing then PC = relative address.

/* BVC */
    if (!IF_OVERFLOW()) {
	clk += ((PC & 0xFF00) != (REL_ADDR(PC, src) & 0xFF00) ? 2 : 1);
	PC = REL_ADDR(PC, src);
    }
