# 6502 Microprocessor - INY pseudocode: YR = (YR + 1) & 0xff; update Sign and Zero.

/* INY */
    unsigned src = YR;
    src = (src + 1) & 0xff;
    SET_SIGN(src);
    SET_ZERO(src);
    YR = (src);
