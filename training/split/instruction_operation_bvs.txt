# 6502 Microprocessor - BVS pseudocode: branch if Overflow flag set; handle page crossing cycles then PC = relative address.

/* BVS */
    if (IF_OVERFLOW()) {
	clk += ((PC & 0xFF00) != (REL_ADDR(PC, src) & 0xFF00) ? 2 : 1);
	PC = REL_ADDR(PC, src);
    }
