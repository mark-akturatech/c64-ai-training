# 6502 Microprocessor - BEQ pseudocode: if Zero flag set then adjust cycles for page crossing and PC to relative target.

/* BEQ */
    if (IF_ZERO()) {
	clk += ((PC & 0xFF00) != (REL_ADDR(PC, src) & 0xFF00) ? 2 : 1);
	PC = REL_ADDR(PC, src);
    }
